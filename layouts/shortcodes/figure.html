{{/* ABOUTME: Custom figure shortcode that uses Hugo's assets pipeline for image processing */}} {{/* ABOUTME: Supports src, alt, caption, link parameters and generates responsive WebP images */}} {{ $src := .Get "src" }} {{ $alt := .Get "alt" }} {{ $caption := .Get "caption" }} {{ $link := .Get "link" }} {{ if $src }} {{ $image := false }} {{ $isResource := false }} {{ if hasPrefix $src "/images/" }} {{ $imagePath := strings.TrimPrefix "/images/" $src }} {{ $imageResource := resources.Get (printf "images/%s" $imagePath) }} {{ if $imageResource }} {{ $image = $imageResource.Resize "800x webp q85" }} {{ $isResource = true }} {{ else }} {{ warnf "Image not found in assets: images/%s, using source path as fallback" $imagePath }} {{ $image = $src }} {{ end }} {{ else }} {{ $image = $src }} {{ end }} {{ end }}
